### Body

<ParamField body="uids" type="array" required>
  An array of user ids to search for
</ParamField>

<ParamField body="offset" type="number" initialValue="0">
  The offset to start searching from
</ParamField>

<ParamField body="limit" type="number" initialValue="10">
  The number of users to return
</ParamField>
